print("3. Does playing at home provide an advantage in creating scoring opportunities?")

xG_H, xG_A = [], []
gospodarze, goscie = [], []

for i in range(380):
    xG_H.append(sezon2324[i]["main_stats"][13]["H"])
    xG_A.append(sezon2324[i]["main_stats"][13]["A"])
    gospodarze.append(sezon2324[i]["przebieg_meczu"][0]["HT"])
    goscie.append(sezon2324[i]["przebieg_meczu"][0]["AT"])

xG_df = pd.DataFrame({
    'xG_gospodarze': xG_H,
    'druzyna_gospodarzy': gospodarze,
    'xG_goscie': xG_A,
    'druzyna_gosci': goscie
})

dane_do_wykresu3_gospodarze = xG_df.groupby('druzyna_gospodarzy').agg(
    srednia_xG_H=('xG_gospodarze', lambda x: round(x.mean(), 2))
).reset_index().sort_values('druzyna_gospodarzy')

dane_do_wykresu3_goscie = xG_df.groupby('druzyna_gosci').agg(
    srednia_xG_A=('xG_goscie', lambda x: round(x.mean(), 2))
).reset_index().sort_values('druzyna_gosci')

dane_do_wykresu3 = pd.merge(dane_do_wykresu3_gospodarze, dane_do_wykresu3_goscie,
                            left_on='druzyna_gospodarzy', right_on='druzyna_gosci')
dane_do_wykresu3 = dane_do_wykresu3.rename(columns={'druzyna_gospodarzy': 'druzyna'})
dane_do_wykresu3['roznica'] = dane_do_wykresu3['srednia_xG_H'] - dane_do_wykresu3['srednia_xG_A']
dane_do_wykresu3 = dane_do_wykresu3.sort_values('srednia_xG_H', ascending=True)

           xmax=dane_do_wykresu3['srednia_xG_H'], color='#b2b2b2', linewidth=4)
plt.scatter(dane_do_wykresu3['srednia_xG_H'], dane_do_wykresu3['druzyna'], color='#2E8B57', s=100, label='Home', zorder=3)
plt.scatter(dane_do_wykresu3['srednia_xG_A'], dane_do_wykresu3['druzyna'], color='#CD5C5C', s=100, label='Away', zorder=3)

plt.title("Does playing at home increase xG?\n2023/2024 Season", fontweight="bold", fontsize=16)
plt.xlabel("Average xG", fontweight="bold", fontsize=12)
plt.ylabel("Team", fontweight="bold", fontsize=12)
plt.legend(title="Venue")

print("5. Which team plays the most aggressively, and which the smartest?")

zolte_H, czerwone_H, zolte_A, czerwone_A = [], [], [], []

    zolte_H.append(sezon2324[i]['main_stats'][7]["H"])

    zolte_A.append(sezon2324[i]['main_stats'][7]["A"])
    czerwone_A.append(sezon2324[i]['main_stats'][8]["A"])

faule_kartki_H = pd.DataFrame({
    'gospodarze': gospodarze,
    'faule_H': np.array(faule_H, dtype=float),
    'zolte_H': zolte_H,
    'czerwone_H': czerwone_H
})

faule_kartki_A = pd.DataFrame({
    'goscie': goscie,
    'faule_A': np.array(faule_A, dtype=float),
    'zolte_A': zolte_A,
    'czerwone_A': czerwone_A
})

print("6. Which team is most dependent on points earned at home?")

punkty_boisko = pd.DataFrame({
    'gospodarze': gospodarze,
    'gole_H': np.array(gole_H, dtype=float),
    'goscie': goscie,
    'gole_A': np.array(gole_A, dtype=float)
})

cond_H = [punkty_boisko['gole_H'] > punkty_boisko['gole_A'], punkty_boisko['gole_H'] == punkty_boisko['gole_A']]
punkty_boisko['punkty_H'] = np.select(cond_H, [3, 1], default=0)

cond_A = [punkty_boisko['gole_A'] > punkty_boisko['gole_H'], punkty_boisko['gole_H'] == punkty_boisko['gole_A']]
punkty_boisko['punkty_A'] = np.select(cond_A, [3, 1], default=0)

punkty_dom = punkty_boisko.groupby('gospodarze')['punkty_H'].sum().reset_index()
punkty_wyjazd = punkty_boisko.groupby('goscie')['punkty_A'].sum().reset_index()

punkty_zdobyte = pd.merge(punkty_dom, punkty_wyjazd, left_on='gospodarze', right_on='goscie')
punkty_zdobyte = punkty_zdobyte.rename(columns={'gospodarze': 'druzyna'}).drop('goscie', axis=1)

punkty_zdobyte['suma_punkty'] = punkty_zdobyte['punkty_H'] + punkty_zdobyte['punkty_A']
punkty_zdobyte['procent_H'] = punkty_zdobyte['punkty_H'] / punkty_zdobyte['suma_punkty']
punkty_zdobyte = punkty_zdobyte.sort_values('procent_H', ascending=False)

punkty_zdobyte_disp = punkty_zdobyte.copy()
punkty_zdobyte_disp['procent_H'] = punkty_zdobyte_disp['procent_H'].apply(lambda x: f"{x:.2%}")
